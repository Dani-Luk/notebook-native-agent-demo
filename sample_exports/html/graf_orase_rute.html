import pandas as pd
import networkx as nx
import matplotlib.pyplot as plt
from IPython.display import display


def build_city_graph(routes_df: pd.DataFrame) -> nx.Graph:
    """Build an undirected weighted graph from a routes dataframe."""
    graph = nx.Graph()
    for _, row in routes_df.iterrows():
        graph.add_edge(row["source"], row["target"], distance=row["distance_km"])
    return graph


def compute_shortest_routes(graph: nx.Graph, start_city: str) -> pd.DataFrame:
    """Compute shortest weighted routes from the selected start city."""
    if start_city not in graph:
        raise ValueError(f"Orașul '{start_city}' nu există în graf.")

    path_lengths = nx.single_source_dijkstra_path_length(graph, start_city, weight="distance")
    paths = nx.single_source_dijkstra_path(graph, start_city, weight="distance")

    rows = []
    for city, distance in path_lengths.items():
        rows.append(
            {
                "destination": city,
                "distance_km": distance,
                "path": " -> ".join(paths[city]),
                "stops": max(len(paths[city]) - 1, 0),
            }
        )

    shortest_paths_df = pd.DataFrame(rows).sort_values(["distance_km", "destination"]).reset_index(drop=True)
    return shortest_paths_df


def plot_city_graph(graph: nx.Graph, start_city: str):
    """Plot the city graph and highlight the start city."""
    fig, ax = plt.subplots(figsize=(10, 7))
    pos = nx.spring_layout(graph, seed=42)

    node_colors = ["tomato" if node == start_city else "skyblue" for node in graph.nodes()]
    edge_labels = nx.get_edge_attributes(graph, "distance")

    nx.draw_networkx(
        graph,
        pos=pos,
        ax=ax,
        with_labels=True,
        node_color=node_colors,
        node_size=1800,
        font_size=10,
        edge_color="gray",
        width=2,
    )
    nx.draw_networkx_edge_labels(
        graph,
        pos=pos,
        edge_labels={edge: f"{dist} km" for edge, dist in edge_labels.items()},
        font_size=9,
        ax=ax,
    )

    ax.set_title(f"Graf orașe și rute (start: {start_city})")
    ax.axis("off")
    return fig

cities_graph = build_city_graph(routes_df)
